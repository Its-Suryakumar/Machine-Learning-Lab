from sklearn.decomposition import PCA
 
n_components = 2
pca = PCA(n_components=n_components)
pca.fit(Z)
x_pca = pca.transform(Z)
print(x_pca.shape)
 
# Create the dataframe
df_pca1 = pd.DataFrame(x_pca,
                       columns=['PC{}'.
                       format(i)
                        for i in range(n_components)])
print(df_pca1)
