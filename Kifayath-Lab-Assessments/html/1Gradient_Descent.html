# Building the model
m = 0
c = 0

L = 0.00000001  #learning Rate
epochs = 100  

n = len(X)

# Gradient Descent 
for i in range(epochs): 
    Y_pred = m*X + c  # predicting Y using m and c
    D_m = (-2/n) * sum(X * (Y - Y_pred))  # Derivative wrt m
    D_c = (-2/n) * sum(Y - Y_pred)  # Derivative wrt c
    m = m - L * D_m
    c = c - L * D_c

print (m, c)
